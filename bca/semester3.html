import java.util.Scanner;
class CharacteriticsofNumber{
	public static void main(String[] args) {
		int ones = 0 , coeffieicent = 0,number;
		Scanner input = new Scanner(System.in);
		System.out.println("Enter a Number : ");
		number = input.nextInt();
		ones = number%10;
		coeffieicent++;
		int num1 = number/10;
		if (num1!=0) {
			int tens = num1%10;
			coeffieicent++;
			System.out.println("The ten's digit of the number is "+tens);
		}
		System.out.println("The one's digit of the number is "+ones);
		System.out.println("The number of coeffieicent is "+coeffieicent);
		
		// to check the number is odd or even
		if(number>0){
			System.out.println("The number "+number +" is positive.");
		}
		else{
			System.out.print("the number " +number +" is negative number.");
		}

	}
}
